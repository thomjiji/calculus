    \begin{equation*} % Use this environment for unnumbered display equations
        \frac{d}{dx} [\cot (x)] =
        \frac{d}{dx}
        \left[
            \frac{\cos(x)}{\sin(x)}
            \right] =
        \frac{
            -\sin(x) \cdot \sin(x) - \cos(x) \cdot \cos(x)
        }{
            \sin^2(x)
        } =
        -\frac{1}{\sin^2(x)} = -\csc^2(x)
    \end{equation*}
